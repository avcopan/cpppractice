314  inline TensorEvaluator(const XprType& op, const Device& device)
315  : m_impl(op.expression(), device), m_device(device), m_dimensions(op.sizes()), m_offsets(op.startIndices())
316  {
317  for (std::size_t i = 0; i < internal::array_size<Dimensions>::value; ++i) {
318  eigen_assert(m_impl.dimensions()[i] >= op.sizes()[i] + op.startIndices()[i]);
319  }
320 
321  const typename TensorEvaluator<ArgType, Device>::Dimensions& input_dims = m_impl.dimensions();
322  const Sizes& output_dims = op.sizes();
323  if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
324  m_inputStrides[0] = 1;
325  for (int i = 1; i < NumDims; ++i) {
326  m_inputStrides[i] = m_inputStrides[i-1] * input_dims[i-1];
327  }
328 
329  // Don't initialize m_fastOutputStrides[0] since it won't ever be accessed.
330  m_outputStrides[0] = 1;
331  for (int i = 1; i < NumDims; ++i) {
332  m_outputStrides[i] = m_outputStrides[i-1] * output_dims[i-1];
333  m_fastOutputStrides[i] = internal::TensorIntDivisor<Index>(m_outputStrides[i]);
334  }
335  } else {
336  m_inputStrides[NumDims-1] = 1;
337  for (int i = NumDims - 2; i >= 0; --i) {
338  m_inputStrides[i] = m_inputStrides[i+1] * input_dims[i+1];
339  }
340 
341  // Don't initialize m_fastOutputStrides[NumDims-1] since it won't ever be accessed.
342  m_outputStrides[NumDims-1] = 1;
343  for (int i = NumDims - 2; i >= 0; --i) {
344  m_outputStrides[i] = m_outputStrides[i+1] * output_dims[i+1];
345  m_fastOutputStrides[i] = internal::TensorIntDivisor<Index>(m_outputStrides[i]);
346  }
347  }
348  }

359  inline bool evalSubExprsIfNeeded(CoeffReturnType* data) {
360  m_impl.evalSubExprsIfNeeded(NULL);
361  if (!NumTraits<typename internal::remove_const<Scalar>::type>::RequireInitialization && data && m_impl.data()) {
362  Index contiguous_values = 1;
363  if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
364  for (int i = 0; i < NumDims; ++i) {
365  contiguous_values *= dimensions()[i];
366  if (dimensions()[i] != m_impl.dimensions()[i]) {
367  break;
368  }
369  }
370  } else {
371  for (int i = NumDims-1; i >= 0; --i) {
372  contiguous_values *= dimensions()[i];
373  if (dimensions()[i] != m_impl.dimensions()[i]) {
374  break;
375  }
376  }
377  }
378  // Use memcpy if it's going to be faster than using the regular evaluation.
379  const MemcpyTriggerForSlicing<Index, Device> trigger(m_device);
380  if (trigger(contiguous_values)) {
381  Scalar* src = (Scalar*)m_impl.data();
382  for (int i = 0; i < internal::array_prod(dimensions()); i += contiguous_values) {
383  Index offset = srcCoeff(i);
384  m_device.memcpy((void*)(data+i), src+offset, contiguous_values * sizeof(Scalar));
385  }
386  return false;
387  }
388  }
389  return true;
390  }

401  template<int LoadMode>
402  inline PacketReturnType packet(Index index) const
403  {
404  const int packetSize = internal::unpacket_traits<PacketReturnType>::size;
405  static_assert(packetSize > 1, "you made a programming mistake");
406  eigen_assert(index+packetSize-1 < internal::array_prod(dimensions()));
407 
408  Index inputIndices[] = {0, 0};
409  Index indices[] = {index, index + packetSize - 1};
410  if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
411  for (int i = NumDims - 1; i > 0; --i) {
412  const Index idx0 = indices[0] / m_fastOutputStrides[i];
413  const Index idx1 = indices[1] / m_fastOutputStrides[i];
414  inputIndices[0] += (idx0 + m_offsets[i]) * m_inputStrides[i];
415  inputIndices[1] += (idx1 + m_offsets[i]) * m_inputStrides[i];
416  indices[0] -= idx0 * m_outputStrides[i];
417  indices[1] -= idx1 * m_outputStrides[i];
418  }
419  inputIndices[0] += (indices[0] + m_offsets[0]);
420  inputIndices[1] += (indices[1] + m_offsets[0]);
421  } else {
422  for (int i = 0; i < NumDims - 1; ++i) {
423  const Index idx0 = indices[0] / m_fastOutputStrides[i];
424  const Index idx1 = indices[1] / m_fastOutputStrides[i];
425  inputIndices[0] += (idx0 + m_offsets[i]) * m_inputStrides[i];
426  inputIndices[1] += (idx1 + m_offsets[i]) * m_inputStrides[i];
427  indices[0] -= idx0 * m_outputStrides[i];
428  indices[1] -= idx1 * m_outputStrides[i];
429  }
430  inputIndices[0] += (indices[0] + m_offsets[NumDims-1]);
431  inputIndices[1] += (indices[1] + m_offsets[NumDims-1]);
432  }
433  if (inputIndices[1] - inputIndices[0] == packetSize - 1) {
434  PacketReturnType rslt = m_impl.template packet<Unaligned>(inputIndices[0]);
435  return rslt;
436  }
437  else {
438  typename internal::remove_const<CoeffReturnType>::type values[packetSize];
439  values[0] = m_impl.coeff(inputIndices[0]);
440  values[packetSize-1] = m_impl.coeff(inputIndices[1]);
441  for (int i = 1; i < packetSize-1; ++i) {
442  values[i] = coeff(index+i);
443  }
444  PacketReturnType rslt = internal::pload<PacketReturnType>(values);
445  return rslt;
446  }

587  this->m_impl.template writePacket<StoreMode>(inputIndices[0], x);
588  }
589  else {

592  this->m_impl.coeffRef(inputIndices[0]) = values[0];
593  this->m_impl.coeffRef(inputIndices[1]) = values[packetSize-1];
594  for (int i = 1; i < packetSize-1; ++i) {
595  this->coeffRef(index+i) = values[i];
596  }
597  }
598  }

601  {

603  for (int i = 0; i < NumDims; ++i) {
604  inputCoords = coords[i] + this->m_offsets[i];
605  }
606  return this->m_impl.coeffRef(inputCoords);
607  }
