212 template <typename T, typename Index, size_t NumDims>
213 class GaussianGenerator {
214  public:
215  static const bool PacketAccess = false;
216 
217  GaussianGenerator(const array<T, NumDims>& means,
218  const array<T, NumDims>& std_devs)
219  : m_means(means)
220  {
221  for (size_t i = 0; i < NumDims; ++i) {
222  m_two_sigmas[i] = std_devs[i] * std_devs[i] * 2;
223  }
224  }
225 
226  T operator()(const array<Index, NumDims>& coordinates) const {
227  T tmp = T(0);
228  for (size_t i = 0; i < NumDims; ++i) {
229  T offset = coordinates[i] - m_means[i];
230  tmp += offset * offset / m_two_sigmas[i];
231  }
232  return std::exp(-tmp);
233  }
234 
235  private:
236  array<T, NumDims> m_means;
237  array<T, NumDims> m_two_sigmas;
238 };
