78 template <DenseIndex Idx>
79 struct tuple_coeff {
80  template <typename... T>
81  static constexpr DenseIndex get(const DenseIndex i, const std::tuple<T...>& t) {
82  return std::get<Idx>(t) * (i == Idx) + tuple_coeff<Idx-1>::get(i, t) * (i != Idx);
83  }
84  template <typename... T>
85  static void set(const DenseIndex i, std::tuple<T...>& t, const DenseIndex value) {
86  if (i == Idx) {
87  update_value(std::get<Idx>(t), value);
88  } else {
89  tuple_coeff<Idx-1>::set(i, t, value);
90  }
91  }
92 
93  template <typename... T>
94  static constexpr bool value_known_statically(const DenseIndex i, const std::tuple<T...>& t) {
95  return ((i == Idx) & is_compile_time_constant<typename std::tuple_element<Idx, std::tuple<T...> >::type>::value) ||
96  tuple_coeff<Idx-1>::value_known_statically(i, t);
97  }
98 
99  template <typename... T>
100  static constexpr bool values_up_to_known_statically(const std::tuple<T...>& t) {
101  return is_compile_time_constant<typename std::tuple_element<Idx, std::tuple<T...> >::type>::value &&
102  tuple_coeff<Idx-1>::values_up_to_known_statically(t);
103  }
104 
105  template <typename... T>
106  static constexpr bool values_up_to_statically_known_to_increase(const std::tuple<T...>& t) {
107  return is_compile_time_constant<typename std::tuple_element<Idx, std::tuple<T...> >::type>::value &&
108  is_compile_time_constant<typename std::tuple_element<Idx-1, std::tuple<T...> >::type>::value &&
109  std::get<Idx>(t) > std::get<Idx-1>(t) &&
110  tuple_coeff<Idx-1>::values_up_to_statically_known_to_increase(t);
111  }
112 };
