212 template<typename Scalar> struct scalar_hypot_op {
213  EIGEN_EMPTY_STRUCT_CTOR(scalar_hypot_op)
214 // typedef typename NumTraits<Scalar>::Real result_type;
215  EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE const Scalar operator() (const Scalar& _x, const Scalar& _y) const
216  {
217  using std::sqrt;
218  Scalar p, qp;
219  if(_x>_y)
220  {
221  p = _x;
222  qp = _y / p;
223  }
224  else
225  {
226  p = _y;
227  qp = _x / p;
228  }
229  return p * sqrt(Scalar(1) + qp*qp);
230  }
231 };

342 /* NOTE why doing the pset1() in packetOp *is* an optimization ?
343  * indeed it seems better to declare m_other as a Packet and do the pset1() once
344  * in the constructor. However, in practice:
345  * - GCC does not like m_other as a Packet and generate a load every time it needs it
346  * - on the other hand GCC is able to moves the pset1() outside the loop :)

348  * (ICC and gcc 4.4 seems to perform well in both cases, the issue is visible with y = a*x + b*y)
