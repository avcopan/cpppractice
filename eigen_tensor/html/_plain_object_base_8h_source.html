874 template <typename Derived, typename OtherDerived, bool IsVector>
875 struct conservative_resize_like_impl
876 {
877  static void run(DenseBase<Derived>& _this, Index rows, Index cols)
878  {
879  if (_this.rows() == rows && _this.cols() == cols) return;
880  EIGEN_STATIC_ASSERT_DYNAMIC_SIZE(Derived)
881 
882  if ( ( Derived::IsRowMajor && _this.cols() == cols) || // row-major and we change only the number of rows
883  (!Derived::IsRowMajor && _this.rows() == rows) ) // column-major and we change only the number of columns
884  {
885  internal::check_rows_cols_for_overflow<Derived::MaxSizeAtCompileTime>::run(rows, cols);
886  _this.derived().m_storage.conservativeResize(rows*cols,rows,cols);
887  }
888  else
889  {
890  // The storage order does not allow us to use reallocation.
891  typename Derived::PlainObject tmp(rows,cols);
892  const Index common_rows = (std::min)(rows, _this.rows());
893  const Index common_cols = (std::min)(cols, _this.cols());
894  tmp.block(0,0,common_rows,common_cols) = _this.block(0,0,common_rows,common_cols);
895  _this.derived().swap(tmp);
896  }
897  }
898 
899  static void run(DenseBase<Derived>& _this, const DenseBase<OtherDerived>& other)
900  {
901  if (_this.rows() == other.rows() && _this.cols() == other.cols()) return;
902 
903  // Note: Here is space for improvement. Basically, for conservativeResize(Index,Index),
904  // neither RowsAtCompileTime or ColsAtCompileTime must be Dynamic. If only one of the
905  // dimensions is dynamic, one could use either conservativeResize(Index rows, NoChange_t) or
906  // conservativeResize(NoChange_t, Index cols). For these methods new static asserts like
907  // EIGEN_STATIC_ASSERT_DYNAMIC_ROWS and EIGEN_STATIC_ASSERT_DYNAMIC_COLS would be good.
908  EIGEN_STATIC_ASSERT_DYNAMIC_SIZE(Derived)
909  EIGEN_STATIC_ASSERT_DYNAMIC_SIZE(OtherDerived)
910 
911  if ( ( Derived::IsRowMajor && _this.cols() == other.cols()) || // row-major and we change only the number of rows
912  (!Derived::IsRowMajor && _this.rows() == other.rows()) ) // column-major and we change only the number of columns
913  {
914  const Index new_rows = other.rows() - _this.rows();
915  const Index new_cols = other.cols() - _this.cols();
916  _this.derived().m_storage.conservativeResize(other.size(),other.rows(),other.cols());
917  if (new_rows>0)
918  _this.bottomRightCorner(new_rows, other.cols()) = other.bottomRows(new_rows);
919  else if (new_cols>0)
920  _this.bottomRightCorner(other.rows(), new_cols) = other.rightCols(new_cols);
921  }
922  else
923  {
924  // The storage order does not allow us to use reallocation.
925  typename Derived::PlainObject tmp(other);
926  const Index common_rows = (std::min)(tmp.rows(), _this.rows());
927  const Index common_cols = (std::min)(tmp.cols(), _this.cols());
928  tmp.block(0,0,common_rows,common_cols) = _this.block(0,0,common_rows,common_cols);
929  _this.derived().swap(tmp);
930  }
931  }
932 };
