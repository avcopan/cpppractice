344  void resize(const array<Index, NumIndices>& dimensions)
345  {
346  std::size_t i;
347  Index size = Index(1);
348  for (i = 0; i < NumIndices; i++) {
349  internal::check_rows_cols_for_overflow<Dynamic>::run(size, dimensions[i]);
350  size *= dimensions[i];
351  }
352  #ifdef EIGEN_INITIALIZE_COEFFS
353  bool size_changed = size != this->size();
354  m_storage.resize(size, dimensions);
355  if(size_changed) EIGEN_INITIALIZE_COEFFS_IF_THAT_OPTION_IS_ENABLED
356  #else
357  m_storage.resize(size, dimensions);
358  #endif
359  }

398  bool checkIndexRange(const array<Index, NumIndices>& indices) const
399  {
400  using internal::array_apply_and_reduce;
401  using internal::array_zip_and_reduce;
402  using internal::greater_equal_zero_op;
403  using internal::logical_and_op;
404  using internal::lesser_op;

407  // check whether the indices are all >= 0
408  array_apply_and_reduce<logical_and_op, greater_equal_zero_op>(indices) &&
409  // check whether the indices fit in the dimensions
410  array_zip_and_reduce<logical_and_op, lesser_op>(indices, m_storage.dimensions());
