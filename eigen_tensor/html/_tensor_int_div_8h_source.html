82 struct TensorIntDivisor {
83  public:
84  inline TensorIntDivisor() {
85  multiplier = 0;
86  shift1 = 0;
87  shift2 = 0;
88  }

90  // Must have 0 < divider < 2^31. This is relaxed to
91  // 0 < divider < 2^63 when using 64-bit indices on platforms that support
92  // the __uint128_t type.
93  inline TensorIntDivisor(const T divider) {
94  const int N = DividerTraits<T>::N;
95  eigen_assert(static_cast<typename UnsignedTraits<T>::type>(divider) < NumTraits<UnsignedType>::highest()/2);
96  eigen_assert(divider > 0);
97 
98  // fast ln2
99  const int leading_zeros = count_leading_zeros(static_cast<UnsignedType>(divider));
100  int log_div = N - leading_zeros;
101  // if divider is a power of two then log_div is 1 more than it should be.
102  if ((static_cast<typename UnsignedTraits<T>::type>(1) << (log_div-1)) == static_cast<typename UnsignedTraits<T>::type>(divider))
103  log_div--;
104 
105  multiplier = DividerHelper<N, T>::computeMultiplier(log_div, divider);
106  shift1 = log_div > 1 ? 1 : log_div;
107  shift2 = log_div > 1 ? log_div-1 : 0;
108  }
109 
110  // Must have 0 <= numerator. On platforms that dont support the __uint128_t
111  // type numerator should also be less than 2^32-1.
112  inline T divide(const T numerator) const {
113  eigen_assert(static_cast<typename UnsignedTraits<T>::type>(numerator) < NumTraits<UnsignedType>::highest()/2);
114  //eigen_assert(numerator >= 0); // this is implicitly asserted by the line above
115 
116  UnsignedType t1 = muluh(multiplier, numerator);
117  UnsignedType t = (static_cast<UnsignedType>(numerator) - t1) >> shift1;
118  return (t1 + t) >> shift2;
119  }
120 
121  private:
122  typedef typename DividerTraits<T>::type UnsignedType;
123  UnsignedType multiplier;
124  int32_t shift1;
125  int32_t shift2;
126 };

130 // Derived from Hacker's Delight.
131 template <>
132 class TensorIntDivisor<int32_t> {
133  public:
134  inline TensorIntDivisor() {
135  magic = 0;
136  shift = 0;
137  }

144  inline int divide(const int32_t n) const {
145  uint64_t v = static_cast<uint64_t>(magic) * static_cast<uint64_t>(n);
146  return (static_cast<uint32_t>(v >> 32) >> shift);
147  }
148 
149 private:
150  // Compute the magic numbers. See Hacker's Delight section 10 for an in
151  // depth explanation.
152  void calcMagic(int32_t d) {
153  const unsigned two31 = 0x80000000; // 2**31.
154  unsigned ad = d;
155  unsigned t = two31 + (ad >> 31);
156  unsigned anc = t - 1 - t%ad; // Absolute value of nc.
157  int p = 31; // Init. p.
158  unsigned q1 = two31/anc; // Init. q1 = 2**p/|nc|.
159  unsigned r1 = two31 - q1*anc; // Init. r1 = rem(2**p, |nc|).
160  unsigned q2 = two31/ad; // Init. q2 = 2**p/|d|.
161  unsigned r2 = two31 - q2*ad; // Init. r2 = rem(2**p, |d|).
162  unsigned delta = 0;
163  do {
164  p = p + 1;
165  q1 = 2*q1; // Update q1 = 2**p/|nc|.
166  r1 = 2*r1; // Update r1 = rem(2**p, |nc|).
167  if (r1 >= anc) { // (Must be an unsigned
168  q1 = q1 + 1; // comparison here).
169  r1 = r1 - anc;}
170  q2 = 2*q2; // Update q2 = 2**p/|d|.
171  r2 = 2*r2; // Update r2 = rem(2**p, |d|).
172  if (r2 >= ad) { // (Must be an unsigned
173  q2 = q2 + 1; // comparison here).
174  r2 = r2 - ad;}
175  delta = ad - r2;
176  } while (q1 < delta || (q1 == delta && r1 == 0));
177 
178  magic = (unsigned)(q2 + 1);
179  shift = p - 32;
180  }
181 
182  uint32_t magic;
183  int32_t shift;
184 };
